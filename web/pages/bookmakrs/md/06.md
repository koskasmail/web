# 🟦 **C. Recursive Folder Renderer**

Replace your entire `renderFolders()` with this:

```js
function renderFolders() {
    const list = document.getElementById("folderList");
    list.innerHTML = "";

    function renderLevel(folders, indent = 0, parentIndex = null) {
        folders.forEach((folder, index) => {
            const div = document.createElement("div");
            div.className = "folder";
            div.style.paddingLeft = (indent * 20) + "px";

            const name = document.createElement("span");
            name.textContent = folder.name;
            name.onclick = () => {
                selectedFolderIndex = { parentIndex, index, levelFolders: folders };
                renderBookmarks();
            };

            const renameBtn = createBtn("R", () => {
                const newName = prompt("Rename folder:", folder.name);
                if (newName) {
                    folder.name = newName;
                    saveData();
                    renderFolders();
                }
            });

            const subBtn = createBtn("+", () => {
                folder.subfolders.push({
                    name: "New Subfolder",
                    bookmarks: [],
                    subfolders: []
                });
                saveData();
                renderFolders();
            });

            const delBtn = createBtn("X", () => {
                if (confirm("Delete folder?")) {
                    folders.splice(index, 1);
                    selectedFolderIndex = null;
                    saveData();
                    renderFolders();
                    renderBookmarks();
                }
            });

            div.append(name, renameBtn, subBtn, delBtn);
            list.appendChild(div);

            // RECURSE INTO SUBFOLDERS
            if (folder.subfolders.length > 0) {
                renderLevel(folder.subfolders, indent + 1, index);
            }
        });
    }

    renderLevel(data.folders);
}
```

# 🟦 **D. Bookmark Rendering Update**

We must locate the correct folder (top‑level or subfolder).  
Replace your `renderBookmarks()` with:

```js
function renderBookmarks() {
    const list = document.getElementById("bookmarkList");
    list.innerHTML = "";

    if (!selectedFolderIndex) return;

    const { parentIndex, index, levelFolders } = selectedFolderIndex;
    const folder = levelFolders[index];

    folder.bookmarks.forEach((bm, i) => {
        const div = document.createElement("div");
        div.className = "bookmark";

        const info = document.createElement("div");
        info.className = "bookmark-info";

        const linkText = bm.url.length > 30 ? bm.name : bm.url;

        info.innerHTML = `<b>${bm.name}</b><br>
                          <a href="${bm.url}" target="_blank">${linkText}</a><br>
                          <small>${bm.comment}</small>`;

        const editBtn = createBtn("Edit", () => {
            const name = prompt("Name:", bm.name);
            const url = prompt("URL:", bm.url);
            const comment = prompt("Comment:", bm.comment);
            if (name && url) {
                bm.name = name;
                bm.url = url;
                bm.comment = comment;
                saveData();
                renderBookmarks();
            }
        });

        const delBtn = createBtn("X", () => {
            if (confirm("Delete bookmark?")) {
                folder.bookmarks.splice(i, 1);
                saveData();
                renderBookmarks();
            }
        });

        div.append(info, editBtn, delBtn);
        list.appendChild(div);
    });
}
```
