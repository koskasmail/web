```html
<script>
  let eventsData = {};
  let currentYear = new Date().getFullYear();

  const monthNames = [
    "January","February","March","April","May","June",
    "July","August","September","October","November","December"
  ];

  const weekdayNames = ["Sun","Mon","Tue","Wed","Thu","Fri","Sat"];

  const yearSelect = document.createElement("select");
  yearSelect.id = "yearSelect";
  yearSelect.style.marginRight = "10px";

  document.querySelector(".year-pill").appendChild(yearSelect);

  const monthSelect = document.getElementById("monthSelect");
  const calendarGrid = document.getElementById("calendarGrid");
  const eventsList = document.getElementById("eventsList");
  const eventsTitle = document.getElementById("eventsTitle");

  async function loadEvents() {
    const response = await fetch("events.json");
    eventsData = await response.json();

    populateYearSelect();
    updateCalendar();
  }

  function populateYearSelect() {
    yearSelect.innerHTML = "";

    Object.keys(eventsData).forEach(year => {
      const opt = document.createElement("option");
      opt.value = year;
      opt.textContent = year;
      yearSelect.appendChild(opt);
    });

    yearSelect.value = currentYear in eventsData ? currentYear : Object.keys(eventsData)[0];
  }

  function updateCalendar() {
    const year = parseInt(yearSelect.value, 10);
    const month = parseInt(monthSelect.value, 10);

    buildCalendar(year, month);
    showAllEvents(year);
  }

  function initMonthSelect() {
    monthNames.forEach((m, i) => {
      const opt = document.createElement("option");
      opt.value = i;
      opt.textContent = m;
      monthSelect.appendChild(opt);
    });

    monthSelect.value = new Date().getMonth();
  }

  function buildCalendar(year, monthIndex) {
    calendarGrid.innerHTML = "";

    weekdayNames.forEach(d => {
      const div = document.createElement("div");
      div.className = "weekday";
      div.textContent = d;
      calendarGrid.appendChild(div);
    });

    const firstDay = new Date(year, monthIndex, 1);
    const startWeekday = firstDay.getDay();
    const daysInMonth = new Date(year, monthIndex + 1, 0).getDate();

    const events = eventsData[year] || [];

    for (let i = 0; i < startWeekday; i++) {
      const div = document.createElement("div");
      div.className = "day empty";
      calendarGrid.appendChild(div);
    }

    for (let day = 1; day <= daysInMonth; day++) {
      const dateStr = `${year}-${String(monthIndex+1).padStart(2,"0")}-${String(day).padStart(2,"0")}`;
      const dayEvents = events.filter(e => e.date === dateStr);

      const div = document.createElement("div");
      div.className = "day";

      const num = document.createElement("div");
      num.className = "day-number";
      num.textContent = day;
      div.appendChild(num);

      const dots = document.createElement("div");
      if (dayEvents.some(e => e.type === "holiday")) {
        const dot = document.createElement("div");
        dot.className = "event-dot holiday";
        dots.appendChild(dot);
      }
      if (dayEvents.some(e => e.type === "music")) {
        const dot = document.createElement("div");
        dot.className = "event-dot music";
        dots.appendChild(dot);
      }
      div.appendChild(dots);

      div.addEventListener("click", () => showEventsForDate(year, dateStr));

      calendarGrid.appendChild(div);
    }
  }

  function showEventsForDate(year, dateStr) {
    const list = eventsData[year].filter(e => e.date === dateStr);
    const d = new Date(dateStr);
    eventsTitle.textContent = `Events ¬∑ ${d.toDateString()}`;
    renderEvents(list);
  }

  function showAllEvents(year) {
    eventsTitle.textContent = `Events ¬∑ ${year}`;
    renderEvents(eventsData[year] || []);
  }

  function renderEvents(list) {
    eventsList.innerHTML = "";

    if (list.length === 0) {
      const div = document.createElement("div");
      div.className = "empty-state";
      div.textContent = "No events for this selection.";
      eventsList.appendChild(div);
      return;
    }

    list.forEach(ev => {
      const card = document.createElement("div");
      card.className = "event-card";

      const header = document.createElement("div");
      header.className = "event-header";

      const title = document.createElement("div");
      title.className = "event-title";
      title.textContent = ev.title;

      badge.className = "event-badge " + ev.type;

      date.className = "event-date";

      desc.className = "event-desc";
      desc.textContent = ev.description;

      card.appendChild(header);
      card.appendChild(date);
      card.appendChild(desc);

      if (ev.ticketUrl) {
        const link = document.createElement("div");
        link.className = "event-link";
        link.innerHTML = `<a href="${ev.ticketUrl}" target="_blank">Buy Tickets</a>`;
        card.appendChild(link);
      }

      eventsList.appendChild(card);
```
